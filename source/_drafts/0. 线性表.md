# 线性表

## 一、线性表的顺序表示

### 线性表的顺序表示（顺序表）的定义

线性表是一种数据元素按线性顺序排列的数据结构。在 Java 中，线性表的顺序表示通常是通过数组实现的。顺序表示的线性表是一种静态数据结构，其中的元素在内存中是连续存储的，可以通过索引快速访问。

在顺序表中，数据元素在逻辑上是连续的，而在物理存储上，它们通常是保存在一段连续的内存空间中。顺序表支持的基本操作包括：

- **插入**：在表中某个位置插入元素。
- **删除**：删除表中某个位置的元素。
- **查找**：查找表中某个位置的元素。
- **更新**：更新表中某个位置的元素。

#### 1.顺序表类定义

```java
public class SeqList<T> {
    private Object[] data;  // 用数组来存储线性表的数据
    private int size;       // 当前顺序表的大小
    private int capacity;   // 顺序表的容量

    // 构造方法，初始化顺序表的容量
    public SeqList(int capacity) {
        this.capacity = capacity;
        this.data = new Object[capacity];
        this.size = 0;
    }

    // 判断顺序表是否为空
    public boolean isEmpty() {
        return size == 0;
    }

    // 判断顺序表是否已满
    public boolean isFull() {
        return size == capacity;
    }

    // 获取顺序表的大小
    public int size() {
        return size;
    }

    // 插入元素到顺序表的指定位置
    public boolean insert(int index, T element) {
        if (index < 0 || index > size) {
            System.out.println("Index out of bounds");
            return false;
        }
        if (size == capacity) {
            System.out.println("List is full");
            return false;
        }

        // 从后向前移动元素，给新元素腾出位置
        for (int i = size; i > index; i--) {
            data[i] = data[i - 1];
        }

        // 插入新元素
        data[index] = element;
        size++;
        return true;
    }

    // 删除指定位置的元素
    public boolean delete(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }

        // 从前向后移动元素，填补删除的元素位置
        for (int i = index; i < size - 1; i++) {
            data[i] = data[i + 1];
        }

        size--;
        return true;
    }

    // 查找指定位置的元素
    public T get(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return null;
        }
        return (T) data[index];
    }

    // 更新指定位置的元素
    public boolean update(int index, T element) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        data[index] = element;
        return true;
    }

    // 打印顺序表中的所有元素
    public void print() {
        for (int i = 0; i < size; i++) {
            System.out.print(data[i] + " ");
        }
        System.out.println();
    }
}

```

#### 2. 使用顺序表

```java
public class Main {
    public static void main(String[] args) {
        // 创建一个顺序表，容量为 5
        SeqList<Integer> list = new SeqList<>(5);

        // 插入元素
        list.insert(0, 10);  // 在位置 0 插入 10
        list.insert(1, 20);  // 在位置 1 插入 20
        list.insert(1, 30);  // 在位置 1 插入 30
        list.insert(3, 40);  // 在位置 3 插入 40
        list.insert(4, 50);  // 在位置 4 插入 50

        System.out.println("顺序表内容：");
        list.print();  // 输出顺序表

        // 获取指定位置的元素
        System.out.println("获取位置 2 的元素：" + list.get(2));

        // 更新元素
        list.update(2, 100);  // 更新位置 2 的元素为 100
        System.out.println("更新位置 2 后的顺序表内容：");
        list.print();

        // 删除元素
        list.delete(1);  // 删除位置 1 的元素
        System.out.println("删除位置 1 后的顺序表内容：");
        list.print();

        // 尝试在无效位置插入
        list.insert(10, 60);  // 插入位置越界
    }
}
```

#### 3. 实现动态扩容

```java
public class SeqList<T> {
    private Object[] data;  // 用数组来存储线性表的数据
    private int size;       // 当前顺序表的大小
    private int capacity;   // 顺序表的容量

    // 构造方法，初始化顺序表的容量
    public SeqList(int initialCapacity) {
        this.capacity = initialCapacity;
        this.data = new Object[capacity];
        this.size = 0;
    }

    // 判断顺序表是否为空
    public boolean isEmpty() {
        return size == 0;
    }

    // 判断顺序表是否已满
    public boolean isFull() {
        return size == capacity;
    }

    // 获取顺序表的大小
    public int size() {
        return size;
    }

    // 动态扩容
    private void resize() {
        // 扩容为原来的 2 倍
        int newCapacity = capacity * 2;
        Object[] newData = new Object[newCapacity];
        System.arraycopy(data, 0, newData, 0, size); // 将旧数据复制到新数组
        data = newData; // 重新指向新的数组
        capacity = newCapacity; // 更新容量
        System.out.println("Array resized to " + newCapacity);
    }

    // 插入元素到顺序表的指定位置
    public boolean insert(int index, T element) {
        if (index < 0 || index > size) {
            System.out.println("Index out of bounds");
            return false;
        }

        // 如果顺序表已满，进行扩容
        if (size == capacity) {
            resize();
        }

        // 从后向前移动元素，给新元素腾出位置
        for (int i = size; i > index; i--) {
            data[i] = data[i - 1];
        }

        // 插入新元素
        data[index] = element;
        size++;
        return true;
    }

    // 删除指定位置的元素
    public boolean delete(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }

        // 从前向后移动元素，填补删除的元素位置
        for (int i = index; i < size - 1; i++) {
            data[i] = data[i + 1];
        }

        size--;
        return true;
    }

    // 查找指定位置的元素
    public T get(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return null;
        }
        return (T) data[index];
    }

    // 更新指定位置的元素
    public boolean update(int index, T element) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        data[index] = element;
        return true;
    }

    // 打印顺序表中的所有元素
    public void print() {
        for (int i = 0; i < size; i++) {
            System.out.print(data[i] + " ");
        }
        System.out.println();
    }
}
```

## 二、线性表的链式表示

### 1. 单链表

#### a. 定义链表结点Node

```java
class Node<T> {
    T data;       // 存储数据
    Node<T> next; // 指向下一个节点的引用
  
    public Node(T data) {
        this.data = data;
        this.next = null;
    }
}

```

#### b. 定义链表类

```java
public class LinkedList<T> {
    private Node<T> head;  // 链表的头节点
    private int size;      // 链表的大小
  
    public LinkedList() {
        head = null; // 初始化时链表为空
        size = 0;
    }
  
    // 获取链表的大小
    public int size() {
        return size;
    }
  
    // 判断链表是否为空
    public boolean isEmpty() {
        return size == 0;
    }
  
    // 在链表头插入元素
    public void insertAtHead(T data) {
        Node<T> newNode = new Node<>(data);
        newNode.next = head;
        head = newNode;
        size++;
    }
  
    // 在链表尾插入元素
    public void insertAtTail(T data) {
        Node<T> newNode = new Node<>(data);
        if (head == null) {
            head = newNode;
        } else {
            Node<T> current = head;
            while (current.next != null) {
                current = current.next;
            }
            current.next = newNode;
        }
        size++;
    }
  
    // 在指定位置插入元素
    public void insertAtIndex(int index, T data) {
        if (index < 0 || index > size) {
            System.out.println("Index out of bounds");
            return;
        }
        if (index == 0) {
            insertAtHead(data);
            return;
        }
        Node<T> newNode = new Node<>(data);
        Node<T> current = head;
        for (int i = 0; i < index - 1; i++) {
            current = current.next;
        }
        newNode.next = current.next;
        current.next = newNode;
        size++;
    }
  
    // 删除指定位置的元素
    public boolean deleteAtIndex(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        if (index == 0) {
            head = head.next;
        } else {
            Node<T> current = head;
            for (int i = 0; i < index - 1; i++) {
                current = current.next;
            }
            current.next = current.next.next;
        }
        size--;
        return true;
    }
  
    // 查找指定位置的元素
    public T get(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return null;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        return current.data;
    }
  
    // 更新指定位置的元素
    public boolean update(int index, T data) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        current.data = data;
        return true;
    }
  
    // 打印链表的所有元素
    public void print() {
        Node<T> current = head;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.next;
        }
        System.out.println();
    }
}

```

### 2. 双链表

#### a. 定义双链表节点类 `Node`

```java
class Node<T> {
    T data;       // 存储数据
    Node<T> next; // 指向下一个节点
    Node<T> prev; // 指向前一个节点
  
    // 构造方法
    public Node(T data) {

    }

```

#### b. 定义双链表类

```java
public class DoublyLinkedList<T> {
    private Node<T> head;  // 链表头节点
    private Node<T> tail;  // 链表尾节点
    private int size;      // 链表的大小
  
    public DoublyLinkedList() {
        head = null;
        tail = null;
        size = 0;
    }
  
    // 获取链表大小
    public int size() {
        return size;
    }
  
    // 判断链表是否为空
    public boolean isEmpty() {
        return size == 0;
    }
  
    // 在链表头插入元素
    public void insertAtHead(T data) {
        Node<T> newNode = new Node<>(data);
        if (head == null) {
            head = newNode;
            tail = newNode; // 如果链表为空，头尾指针指向同一个节点
        } else {
            newNode.next = head;
            head.prev = newNode;
            head = newNode;
        }
        size++;
    }
  
    // 在链表尾插入元素
    public void insertAtTail(T data) {
        Node<T> newNode = new Node<>(data);
        if (tail == null) {
            head = newNode;
            tail = newNode; // 如果链表为空，头尾指针指向同一个节点
        } else {
            tail.next = newNode;
            newNode.prev = tail;
            tail = newNode;
        }
        size++;
    }
  
    // 在指定位置插入元素
    public void insertAtIndex(int index, T data) {
        if (index < 0 || index > size) {
            System.out.println("Index out of bounds");
            return;
        }
        if (index == 0) {
            insertAtHead(data);
        } else if (index == size) {
            insertAtTail(data);
        } else {
            Node<T> newNode = new Node<>(data);
            Node<T> current = head;
            for (int i = 0; i < index; i++) {
                current = current.next;
            }
            newNode.next = current;
            newNode.prev = current.prev;
            current.prev.next = newNode;
            current.prev = newNode;
            size++;
        }
    }
  
    // 删除指定位置的元素
    public boolean deleteAtIndex(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        if (index == 0) {
            if (head == tail) {
                head = tail = null; // 如果链表中只有一个元素
            } else {
                head = head.next;
                head.prev = null;
            }
        } else if (index == size - 1) {
            tail = tail.prev;
            tail.next = null;
        } else {
            Node<T> current = head;
            for (int i = 0; i < index; i++) {
                current = current.next;
            }
            current.prev.next = current.next;
            current.next.prev = current.prev;
        }
        size--;
        return true;
    }
  
    // 查找指定位置的元素
    public T get(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return null;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        return current.data;
    }
  
    // 更新指定位置的元素
    public boolean update(int index, T data) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        current.data = data;
        return true;
    }
  
    // 打印链表的所有元素
    public void print() {
        Node<T> current = head;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.next;
        }
        System.out.println();
    }
  
    // 打印链表从尾到头的元素（反向遍历）
    public void printReverse() {
        Node<T> current = tail;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.prev;
        }
        System.out.println();
    }
}

```

### 3. 循环链表

```java
class Node<T> {
    T data;       // 存储数据
    Node<T> next; // 指向下一个节点

    public Node(T data) {
        this.data = data;
        this.next = null;
    }
}

```

```java
public class CircularLinkedList<T> {
    private Node<T> head;  // 链表头节点
    private Node<T> tail;  // 链表尾节点
    private int size;      // 链表的大小
  
    public CircularLinkedList() {
        head = null;
        tail = null;
        size = 0;
    }

    // 获取链表大小
    public int size() {
        return size;
    }

    // 判断链表是否为空
    public boolean isEmpty() {
        return size == 0;
    }

    // 在链表头插入元素
    public void insertAtHead(T data) {
        Node<T> newNode = new Node<>(data);
        if (head == null) {
            head = newNode;
            tail = newNode; // 如果链表为空，头尾指针都指向新节点
            newNode.next = head; // 循环链表的尾节点指向头节点
        } else {
            newNode.next = head;
            head = newNode;
            tail.next = head; // 确保尾节点指向头节点
        }
        size++;
    }

    // 在链表尾插入元素
    public void insertAtTail(T data) {
        Node<T> newNode = new Node<>(data);
        if (head == null) {
            head = newNode;
            tail = newNode; // 如果链表为空，头尾指针都指向新节点
            newNode.next = head; // 循环链表的尾节点指向头节点
        } else {
            tail.next = newNode;
            tail = newNode;
            tail.next = head; // 尾节点指向头节点
        }
        size++;
    }

    // 在指定位置插入元素
    public void insertAtIndex(int index, T data) {
        if (index < 0 || index > size) {
            System.out.println("Index out of bounds");
            return;
        }
        if (index == 0) {
            insertAtHead(data);
        } else if (index == size) {
            insertAtTail(data);
        } else {
            Node<T> newNode = new Node<>(data);
            Node<T> current = head;
            for (int i = 0; i < index - 1; i++) {
                current = current.next;
            }
            newNode.next = current.next;
            current.next = newNode;
            size++;
        }
    }

    // 删除指定位置的元素
    public boolean deleteAtIndex(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        if (index == 0) {
            if (head == tail) {
                head = null;
                tail = null; // 如果链表中只有一个元素
            } else {
                head = head.next;
                tail.next = head; // 确保尾节点指向新的头节点
            }
        } else {
            Node<T> current = head;
            for (int i = 0; i < index - 1; i++) {
                current = current.next;
            }
            current.next = current.next.next;
            if (index == size - 1) {
                tail = current; // 如果删除的是尾节点，更新尾节点
            }
        }
        size--;
        return true;
    }

    // 查找指定位置的元素
    public T get(int index) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return null;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        return current.data;
    }

    // 更新指定位置的元素
    public boolean update(int index, T data) {
        if (index < 0 || index >= size) {
            System.out.println("Index out of bounds");
            return false;
        }
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        current.data = data;
        return true;
    }

    // 打印链表的所有元素
    public void print() {
        if (head == null) {
            System.out.println("The list is empty.");
            return;
        }
        Node<T> current = head;
        do {
            System.out.print(current.data + " ");
            current = current.next;
        } while (current != head); // 循环直到回到头节点
        System.out.println();
    }
}
```
